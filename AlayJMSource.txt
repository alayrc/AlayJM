const animationStyles = document.createElement('style');
animationStyles.innerHTML = `
  @keyframes text-morph {
    0% {
      transform: scale(1);
      color: red;
    }
    25% {
      transform: scale(1.2) skewX(10deg);
      color: orange;
    }
    50% {
      transform: scale(0.8) skewY(-10deg);
      color: yellow;
    }
    75% {
      transform: scale(1.1) skewX(-10deg);
      color: green;
    }
    100% {
      transform: scale(1);
      color: blue;
    }
  }
`;
document.head.appendChild(animationStyles);
setInterval(() => {
  const newParagraph = document.createElement('p');
  newParagraph.textContent = 'savas is a monkey!';
  newParagraph.style.fontFamily = 'sans-serif';
  newParagraph.style.animation = 'text-morph 5s infinite alternate ease-in-out';
  const randomX = Math.random() * window.innerWidth;
  const randomY = Math.random() * window.innerHeight;
  const randomDelay = Math.random() * 0.5;
  newParagraph.style.position = 'absolute';
  newParagraph.style.left = randomX + 'px';
  newParagraph.style.top = randomY + 'px';
  newParagraph.style.animationDelay = randomDelay + 's';
  document.body.appendChild(newParagraph);
}, 100);
